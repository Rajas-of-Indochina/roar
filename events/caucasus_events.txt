caucasianroa.017 = {
	type = character_event
	hidden = yes
	trigger = {
		exists = father
		father = {
			OR = {
				has_trait = achebeg
				has_trait = achebeg_descendant
			}
		}
		NOR = {
			has_trait = achebeg			# Just in case the tree was horribly tangled...
			has_trait = achebeg_descendant
		}
	}
	immediate = {
		add_trait = achebeg_descendant
		every_child = {
			trigger_event = caucasianroa.017
		}
	}
}
