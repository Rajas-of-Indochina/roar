	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_bakanak
		}
	}
	parameters = {
		language_branch_shaz_turkic = yes
		language_family_turkic = yes
		language_union_steppe_siberian = yes

	ai_will_do = {
		value = 10

			limit = { has_cultural_pillar = language_bakanak }

		}

		else_if = {

		}
		else_if = {
